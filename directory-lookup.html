Directory lookup PHP

Support additional payment methods using directory lookup. Create a skin on the Adyen side to display these methods. 

// initialize service
$service = new Service\DirectoryLookup($client);

$sessionValidity = date(
    DATE_ATOM,
    mktime(date("H") + 10, date("i"), date("s"), date("m"), date("j"), date("Y"))
);

$json = '{
      "paymentAmount": "1000",
      "currencyCode": "EUR"
      "merchantReference": "Get Payment methods",
      "skinCode": "YOUR SKIN CODE"
      "merchantAccount": "YOUR MERCHANT ACCOUNT",
      "sessionValidity": "'.$sessionValidity.'"
      "countryCode": "NL",
      "shopperLocale": "nl_NL"
}';

$params = json_decode($json, true);

// add signature in request
$hmacKey = 'YOUR HMAC KEY OF THE SKIN';
$params["merchantSig"] = Util::calculateSha256Signature($hmacKey, $params);

$result = $service->directoryLookup($params);
